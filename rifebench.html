vspipe

# Command constructed by rifebench:
vspipe -c y4m -p bench.vpy -- \
  -a source=video.mkv \
  -a model=47 \
  -a gpu_id=0 \
  -a format=RGBH \
  -a multi=2 \
  -a streams=2 \
  -a trt_fp16=1 \
  -a use_cuda_graph=1

Output 2882 frames in 29.72 seconds (96.99 fps)

ffmpeg

# vspipe output piped to ffmpeg:
vspipe -c y4m -p bench.vpy - -a source=... | \
  ffmpeg -benchmark -i - -c:v hevc_nvenc -preset p5 -cq 18 \
  -profile:v main10 -pix_fmt p010le -f null NUL
